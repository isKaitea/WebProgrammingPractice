Switch

詳解

        Switch (表達示) {
            case 值1:
                當表達示等於值1時執行
                break;
            case 值2:
                當表達示等於值2時執行
                break;
            case 值3:
                當表達示等於值3時執行
                break;
            ...
            default:
                當表達示不符合任何情況時執行
        }

for 迴圈

        for(初始化計數器;執行條件;步進值){
            //重複執行的程式碼
        }

for/in

針對陣列的項目做處理

        for(變數 in 陣列物件){
            //重複執行的程式碼
        }

while 迴圈

        while(執行的條件){
            // 重複執行的程式碼
        }

do while 迴圈

無論如何最少會跑一次

        do {
            //重複執行的程式碼
        }while(執行的條件);

brack 與 continue

        // break 用來跳出迴圈
        for (let i = 0; i < 10; i++) {
            if (i == 5) {
                break;//當 i 等於5 時跳出迴圈
            }
            // 其他程式碼
        }

        //continue 用來跳過本次迴圈,進入下一次迴圈
        for (let i = 0; i < 10; i++) {
            if (i % 2 = 0) {
                continue;//當i為偶數時,跳過本次迴圈
            }
                // 其他程式碼
        }
